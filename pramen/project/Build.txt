import sbt._
import Keys._

object B extends Build {
  lazy val root =
    Project("root", file("."))
      .configs( FunTest )
      .settings( inConfig(FunTest)(Defaults.testTasks) : _*)
      .settings(
        libraryDependencies += specs,
        testOptions in Test := Seq(Tests.Filter(itFilter)),
        testOptions in FunTest := Seq(Tests.Filter(unitFilter))
      )

  def itFilter(name: String): Boolean = name endsWith "LongSuite"
  def unitFilter(name: String): Boolean = (name endsWith "Suite") && !itFilter(name)

  lazy val FunTest = config("integration") extend(Test)
}
